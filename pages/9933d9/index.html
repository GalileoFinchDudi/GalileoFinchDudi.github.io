    # 配置Babel解决兼容性问题

    由于一些浏览器，嵌入设备都比较老旧，不支持ES6的语法，所以需要将ES6的语法编译成ES5的语法

    pnpm i D @babel/core @babel/preset-env babel-loader core-js
    1

    配置babel







     
     
     
     
     
     
     
     
     
     
     
     
     
     
     
     
     
     
     







    // webpack.config.js
module: {
  rules: [
    {
      test: /\.ts$/,
      use: [
        {
          loader: "babel-loader",
          options: {
            presets: [
              [
                "@babel/preset-env",
                {
                  targets: {
                    chrome: "58",
                    ie: "11",
                  },
                  corejs: "3",
                  // 使用corejs的方式 "usage" 表示按需加载
                  useBuiltIns: "usage",
                },
              ],
            ],
          },
        },
        "ts-loader",
      ],
      exclude: /node_modules/,
    },
  ],
},
